94 // MARE_API_ASSERT
95 // - Throws api_exception.
96 // - Enabled by default; disable with MARE_DISABLE_CHECK_API.
97 // - See debug.hh for macros that control how this is enabled
98 // Used for reporting invalid arguments passed in or returned via the API.
99 #ifdef MARE_CHECK_API
100 #define MARE_API_ASSERT(cond, format, ...) do { \
101  if(!(cond)) { throw \
102  ::mare::api_exception(::mare::internal:: \
103  strprintf(format, ## __VA_ARGS__), \
104  __FILE__, __LINE__, __FUNCTION__); \
105  } } while(false)
106 #else
107 #define MARE_API_ASSERT(cond, format, ...) do { \
108  if(false) { if(cond){} if(format){} } } while(false)
109 #endif // MARE_CHECK_API
110 
111 // MARE_API_DETAILED_ASSERT
112 // - Throws api_exception.
113 // - Disabled by default; enable with MARE_ENABLE_CHECK_API_DETAILED.
114 // - See debug.hh for macros that control how this is enabled
115 // Used for extra detailed and slower checking of the API calls.
116 // For example: circular dependency checks, deadlock detection.
117 #ifdef MARE_CHECK_API_DETAILED
118 #define MARE_API_DETAILED_ASSERT(cond, format, ...) do { if(!(cond)) { throw \
119  ::mare::api_exception(::mare::internal:: \
120  strprintf(format, ## __VA_ARGS__), \
121  __FILE__, __LINE__, __FUNCTION__); } \
122  } while(false)
123 #else
124 #define MARE_API_DETAILED_ASSERT(cond, format, ...) do { if(false) { \
125  if(cond){} if(format){} } } while(false)
