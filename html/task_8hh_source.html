629 inline void launch_and_reset(group_ptr const& group, task_ptr& task)
630 {
631  internal::task* t = internal::c_ptr(task);
632  MARE_API_ASSERT(internal::c_ptr(task), "null task_ptr");
633 
634  // Instead of increasing the task.ref_counter, and then
635  // decreasing it, we keep it as-is.
636  reset_but_not_unref(task);
637  internal::group* g = internal::c_ptr(group);
638  t->launch(g, internal::ref_count_policy::do_nothing);
639 }
