46 template <class T, size_t Alignment>
47 // Inherit construct(), destruct() etc.
48 struct aligned_allocator : public std::allocator<T>
49 {
50 
51 MARE_GCC_IGNORE_END("-Weffc++");
52 
53  typedef typename std::allocator<T>::size_type size_type;
54  typedef typename std::allocator<T>::pointer pointer;
55  typedef typename std::allocator<T>::const_pointer const_pointer;
56 
57  //Defines an aligned allocator suitable for allocating elements of type U.
58  template <class U>
59  struct rebind {
60  typedef aligned_allocator<U,Alignment> other;
61  };
62 
63 
64  //Default-constructs an allocator.
65  aligned_allocator() throw() {}
66 
67 
68  //Copy-constructs an allocator.
69  aligned_allocator(const aligned_allocator& other) throw() :
70  std::allocator<T>(other) {}
71 
72 
73  //Convert-constructs an allocator.
74  template <class U>
75  aligned_allocator(const aligned_allocator<U,Alignment>&) throw() {}
76 
77 
78  //Destroys an allocator.
79  ~aligned_allocator() throw() {}
80 
81 
82  //Allocates n elements of type T, aligned to a multiple of
83  //Alignment.
84  pointer allocate(size_type n) {
85  return allocate(n, const_pointer(0));
86  }
87 
88 
89  //Allocates n elements of type @c T, aligned to a multiple of Alignment.
90  pointer allocate(size_type n, const_pointer) {
91  void *p;
92  p = internal::mare_aligned_malloc(Alignment, n*sizeof(T));
93  if (!p)
94  throw std::bad_alloc();
95 
96  return static_cast<pointer>(p);
97  }
98 
99 
100  //Frees the memory previously allocated by an aligned allocator.
101  void deallocate(pointer p, size_type) {
102  internal::mare_aligned_free(p);
103  }
104 
105 };
