160 inline group_ptr intersect(group_ptr const& a, group_ptr const& b)
161 {
162  using namespace internal;
163 
164  group* a_ptr = c_ptr(a);
165  group* b_ptr = c_ptr(b);
166 
167  // If a group meets nullptr, the result is nullptr.
168  if (a_ptr == nullptr || b_ptr == nullptr )
169  return group_ptr(nullptr);
170 
171  if (a_ptr->is_ancestor_of(b_ptr))
172  return b;
173 
174  if (b_ptr->is_ancestor_of(a_ptr))
175  return a;
176 
177  group_ptr retval(internal::lattice::create_meet_node(a_ptr, b_ptr));
178  return retval;
179 }

305 inline void
306 spin_wait_for(group_ptr const& group)
307 {
308  auto g_ptr = internal::c_ptr(group);
309  MARE_API_ASSERT(g_ptr, "null group_ptr");
310 
311  if (internal::current_task()) {
312  g_ptr->wait(); // not spinning inside of a task to avoid wasting cycles
313  return;
314  }
315 
316  size_t spin_count = 200000;
317  while (!g_ptr->is_empty(std::memory_order_relaxed)) {
318  if (spin_count-- == 0) {
319  g_ptr->wait();
320  return;
321  }
322  }
323 } /* end_addtogroup groups_waiting */
