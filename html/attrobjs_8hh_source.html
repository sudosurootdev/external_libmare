534 template<typename Body, typename = void>

537  typedef Body type;
538  typedef std::false_type result_type;

541 //Way to identify that body is gpu_kernel type
542 //For gpu tasks, Body will be mare_shared_ptr,
543 //we get the type from mare_shared_ptr and
544 //check if type derives from gpu_kernel_base; this
545 //will be true only on the gpu path.
546 template<typename Body>
547 struct is_gpu_kernel<
548  Body,
549  typename is_gpu_kernel<void, typename Body::type>::type>
550 {
551  typedef typename Body::type type;
552  typedef typename std::is_base_of<internal::gpu_kernel_base,
553  typename Body::type>::type result_type;
554 };
