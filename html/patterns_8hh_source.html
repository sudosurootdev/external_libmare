35 template<class T>
36 T static_chunk_size(T count)
37 {
38  // Inexpensive way to create fine-grained tasks; the real solution is
39  // outlined below (adaptable or user-configurable chunking)
40  T const GRANULARITY_MULTIPLIER = 4;
41  return std::max(T(1),
42  count /
43  (GRANULARITY_MULTIPLIER *
44  static_cast<T>(internal::num_execution_contexts())));
45 }

138 template <class InputIterator, typename UnaryFn>
139 void pfor_each_async(group_ptr g, InputIterator first, InputIterator last,
140  UnaryFn fn) {
141  typedef typename internal::distance_helper<InputIterator>::
142  _result_type working_type;
143 
144  if (first >= last)
145  return;
146 
147  working_type const dist = internal::distance(first,last);
148  // blksz could be picked better, and should perhaps also be under
149  // user control. E.g., if fn is expensive, we may want smaller
150  // blocks; Splitting into blocks should possibly be moved elsewhere.
151  // Also, alignment, blk_size, etc., may be dependent on fn.
152  working_type blksz = static_chunk_size<working_type>(dist);
153  MARE_INTERNAL_ASSERT(blksz > 0, "block size should be greater than zero");
154  InputIterator lb = first;
155  while (lb < last) {
156  auto safeofs = std::min(working_type(internal::distance(lb,last)), blksz);
157  InputIterator rb = lb + safeofs;
158 
159  launch(g, [=] {
160  InputIterator it = lb;
161  while (it < rb)
162  fn(it++);
163  });
164  lb = rb;
165  }
166 }

183 template<size_t DIMS, typename UnaryFn>
184 void pfor_each_async(group_ptr g, const mare::range<DIMS>& r, UnaryFn fn)
185 {
186  //range can have offsets, but bounds are always zero based,
187  //mare::range::linear_to_index will take care of adding the offsets
188  //if any in range.
189  pfor_each_async(g, size_t(0), r.size(), [=] (size_t i) {

243 template<typename Body>
244 void
245 pfor_each_sizet(group_ptr group, size_t first, size_t last, Body&& body) {
246 
247  if (first >= last)
248  return ;
249 
250  // Wrap the body so that we get a nice interface to it
251  typedef mare::internal::body_wrapper<Body> pfor_body_wrapper;
252  pfor_body_wrapper wrapper(std::forward<Body>(body));
253 
254  // Let's make sure that the function passed is unary
255  MARE_INTERNAL_ASSERT(pfor_body_wrapper::get_arity() == 1,
256  "invalid number of arguments in pfor body");
257 
258  // Let's get the body attributes, and add pfor to them
259  auto body_attrs = wrapper.get_attrs();
260  body_attrs = mare::add_attr(body_attrs, mare::internal::attr::pfor);
261 
262  // We get a reference to the wrapper
263  auto& fn = wrapper.get_fn();
264  typedef typename std::remove_reference<decltype(fn)>::type UnaryFn;
265 
266  // blk_size defines the minimal granuality of iterations to advance.
267  // TODO: We have to figure out how to pass blk_size to API.
268  const size_t blk_size = 1;
269 
270  // We first check whether this is a nested pfor or not.
271  // It is, we treat it differently depending on whether
272  // the user passed a group pointer to it or not:
273  // - if she didn't, then we execute the pfor as a regular for.
274  // - if she did, we create a task that executes it as a regular
275  // for, but we execute it inline.
276  //
277  auto t = internal::current_task();
278  if (t && t->is_pfor()) {
279 
280  if (group == nullptr) {
281  for (size_t i = first; i < last; ++i)
282  fn(i);
283  return ;
284  }
285 
286  auto inlined = mare::create_task(mare::with_attrs(body_attrs,
287  [first, last, &fn] {
288  for (size_t i = first; i < last; ++i) {
289  fn(i);
290  }
291  }));
292  internal::c_ptr(inlined)->execute_inline(internal::c_ptr(group));
293  return;
294  };
295 
296  // Create the group that will take care of the parallel for
297  auto g_internal = create_group();
298  auto g = g_internal;
299  if(group)
300  g = g & group;
301 
302  typedef internal::adaptive_pfor_strategy<UnaryFn> strategy_type;
303 
304  // adaptive_pfor is in fact the storage with a work steal tree,
305  // the group pointer, and the lambda.
306  strategy_type adaptive_pfor(g, first, last - 1,
307  std::forward<UnaryFn>(fn),
308  body_attrs,
309  blk_size);
310 
311  size_t max_tasks = adaptive_pfor.get_max_tasks();
312  if (max_tasks > 1 && max_tasks < (last - first))
313  adaptive_pfor.static_split(max_tasks);
314 
315  // We are the ones that found the parallel for, so we start
316  // stealing from the pfor tree inline. If the pfor is short
317  // enough, we might not even spawn many new tasks.
318  execute_master_task<strategy_type>(adaptive_pfor);
319 
320  spin_wait_for(g);
321 
322 #ifdef ADAPTIVE_PFOR_DEBUG
323  print_tree(adaptive_pfor.get_tree());
324 #endif // ADAPTIVE_PFOR_DEBUG
325 }

586 template <typename InputIterator, typename OutputIterator, typename UnaryFn>
587 void ptransform(group_ptr group, InputIterator first, InputIterator last,
588  OutputIterator d_first, UnaryFn fn) {
589  // In principle, pfor_each() is sufficient as a building block for all
590  // ptransform() variants, but a direct implementation needs less
591  // iterator frobbing.
592  if (first >= last)
593  return;
594  auto g_internal = create_group("ptransform/4");
595  auto g = g_internal;
596  if (group)
597  g = g & group;
598 
599  size_t const dist = std::distance(first,last);
600  auto blksz = static_chunk_size(dist);
601  MARE_INTERNAL_ASSERT(blksz > 0, "block size should be greater than zero");
602  auto lb = first;
603  auto d_lb = d_first;
604  while (lb < last) {
605  auto safeofs = std::min(size_t(std::distance(lb,last)), blksz);
606  InputIterator rb = lb + safeofs;
607 
608  launch_or_exec(rb != last, g, [=] {
609  InputIterator it = lb;
610  OutputIterator d_it = d_lb;
611  while (it < rb)
612  *d_it++ = fn(*it++);
613  });
614  lb = rb;
615  d_lb += safeofs;
616  }
617 
618  spin_wait_for(g_internal);
619 }

698 template <typename InputIterator, typename OutputIterator, typename BinaryFn>
699 void ptransform(group_ptr group,
700  InputIterator first1, InputIterator last1,
701  InputIterator first2,
702  OutputIterator d_first, BinaryFn fn) {
703  if (first1 >= last1)
704  return;
705  auto g_internal = create_group("ptransform/5");
706  auto g = g_internal;
707  if (group)
708  g = g & group;
709 
710  size_t const dist = std::distance(first1,last1);
711  auto blksz = static_chunk_size(dist);
712  MARE_INTERNAL_ASSERT(blksz > 0, "block size should be greater than zero");
713  auto lb = first1;
714  auto lb2 = first2;
715  auto d_lb = d_first;
716  while (lb < last1) {
717  auto safeofs = std::min(size_t(std::distance(lb,last1)), blksz);
718  InputIterator rb = lb + safeofs;
719 
720  launch_or_exec(rb != last1, g, [=] {
721  InputIterator it = lb;
722  InputIterator it2 = lb2;
723  OutputIterator d_it = d_lb;
724  while (it < rb)
725  *d_it++ = fn(*it++, *it2++);
726  });
727  lb = rb;
728  lb2 += safeofs;
729  d_lb += safeofs;
730  }
731 
732  spin_wait_for(g_internal);
733 }

885 template <typename InputIterator, typename BinaryFn>
886 void pscan_inclusive_rec(group_ptr g, InputIterator first, InputIterator last,
887  BinaryFn fn) {
888  size_t dist = std::distance(first,last);
889  if (dist < 2)
890  return;
891 
892  auto pivot = first + dist/2;
893  pscan_inclusive_rec(g, first, pivot, fn);
894  pscan_inclusive_rec(g, pivot, last, fn);
895 
896  spin_wait_for(g);
897 
898  launch(g, [=] {
899  // like *(pivot-1), but with fwd iters only
900  auto const& lhs = *(first + (dist/2-1));
901  pfor_each_async(g, pivot, last, [=] (InputIterator it) {
902  *it = fn(lhs, *it);
903  });
904  });
905 };
