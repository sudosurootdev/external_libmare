48  T* get() const {
49  auto pvalue = static_cast<T*>(scope_type::get_specific(_key));
50  if (!pvalue) {
51  pvalue = _alloc.allocate(1); // owned by Scope
52  _alloc.construct(pvalue);
53  scope_type::set_specific(_key, pvalue);
54  }
55  return pvalue;
56  }

78  bool operator!() const {

80  pointer_type t = get();

82  this);

84 #else
85  return false;

103  operator bool() const {
104 #ifdef MARE_CHECK_INTERNAL
105  return get() != nullptr;
106 #else
107  return true;

111 private:
112  static void maybe_invoke_dtor(void* ptr) {
113  if (!ptr)
114  return;
115  auto obj = static_cast<T*>(ptr);
116  _alloc.destroy(obj);
117  _alloc.deallocate(obj, 1);
118  }

120  void init_key() {
121  int err = scope_type::key_create(&_key, &maybe_invoke_dtor);
122  if (err != 0)
123  throw mare::tls_exception("Unable to allocate storage key.",
124  __FILE__, __LINE__, __FUNCTION__);
125  }
126 private:
127  internal::storage_key _key; //stores scheduler-local storage key
128  static Allocator _alloc;
129 };
